#include <stdio.h>

#define SIZE 100

void main()
{
    int a[SIZE], b[SIZE], sum[SIZE];
    int *p1, *p2, *p3;
    int n, i;

    printf("Enter the size of arrays: ");
    scanf("%d", &n);

    printf("Enter elements of first array:\n");
    
    for(i = 0; i < n; i++)
    {
        scanf("%d", &a[i]);
    }

    printf("Enter elements of second array:\n");
    
    for(i = 0; i < n; i++)
    {
        scanf("%d", &b[i]);
    }

    p1 = a;
    p2 = b;
    p3 = sum;

    for(i = 0; i < n; i++)
    {
        *(p3 + i) = *(p1 + i) + *(p2 + i);
    }

    printf("Sum of the two arrays:\n");
    
    for(i = 0; i < n; i++)
    {
        printf("%d ", *(p3 + i));
    }
}
