#include <stdio.h>

void main()
{
    char str[100];
    char *ptr;
    int length = 0;

    printf("Enter any string: ");
    fgets(str, sizeof(str), stdin);

    ptr = str;

    while (*ptr != '\0' && *ptr != '\n')
    { 
        length++;
        ptr++;
    }

    printf("The length of a string is: %d\n", length);
}

#include <stdio.h>

int calculateCube(int num)
{
    return num * num * num;
}

void findCubesOf2DArray(int *arr, int rows, int cols)
{
    printf("Cubes of elements in the 2D array:\n");
    for (int i = 0; i < rows; i++)
    {
        for (int j = 0; j < cols; j++)
        {
            int element = *(arr + i * cols + j);
            int cube = calculateCube(element);
            printf("%d ", cube);
        }
        printf("\n");
    }
}

void main()
{
    int rows, cols;

    printf("Enter the number of rows: ");
    scanf("%d", &rows);
    printf("Enter the number of columns: ");
    scanf("%d", &cols);

    int arr[rows][cols];

    printf("Enter elements of the 2D array:\n");
    for (int i = 0; i < rows; i++) {
        for (int j = 0; j < cols; j++) {
            printf("Enter element at [%d][%d]: ", i, j);
            scanf("%d", &arr[i][j]);
        }
    }
    findCubesOf2DArray((int *)arr, rows, cols);
}
