#include <stdio.h>
void main()
{
    int number, count=0;
    
    printf("Enter a number : ");
    scanf("%d", &number);
    
    if(number == 0)
    {
        count = 1;
        printf("%d",count);
    }
    else
    {
        if (number < 0)
        {
            number = -number;
        }
        
        do
        {
            number /= 10;
            count++;
        }
        while(number>0);
        
    }
    printf("%d",count);
    
}
